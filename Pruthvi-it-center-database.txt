>create table personal_info (
  user_id int NOT NULL PRIMARY KEY,
  first_name varchar(50) NOT NULL,
  last_name varchar(25) NOT NULL,
  mobile_no varchar(13) constraint CK_personal_info_mobile_no check (
    mobile_no like '[0-9][0-9][-][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]'
  ) NOT NULL,
  location varchar(25) NOT NULL,
  age char(2));


>show tables;

>create table login_info (
index_id smallint unsigned AUTO_INCREMENT PRIMARY KEY,
username varchar(25) NOT NULL,
password varchar(25) NOT NULL,
user_id char(2),
FOREIGN KEY(user_id) REFERENCES personal_info(user_id)
);

>show tables;

>create table order_history (
  order_id bigint PRIMARY KEY,
  product_id varchar(25),
  product_name varchar(255),
  FOREIGN KEY(product_id) REFERENCES product_table(product_id),
  FOREIGN KEY(product_name) REFERENCES product_table(product_name)
);

>show tables;

>create table product_table (
id bigint NOT NULL,
product_id varchar(25) NOT NULL PRIMARY KEY,
product_name varchar(255) NOT NULL,
product_price varchar(6) NOT NULL,
available_size varchar(5) NOT NULL);


>create table indexing (id int FOREIGN KEY REFERENCES personal_info(id),
username varchar(25) NOT NULL FOREIGN KEY REFERENCES login_info(username));
