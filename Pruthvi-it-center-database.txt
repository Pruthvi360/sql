>create table login_info (

>create table order_history(
order bigint NOT NULL PRIMARY KEY,
product_id varchar(25) FOREIGN KEY REFERENCES product_table(product_id),
product_name varchar(255) FOREIGN KEY REFERENCES product_table(product_name));

>show tables;

>create table product_table (
id bigint NOT NULL,
product_id varchar(25) NOT NULL PRIMARY KEY,
product_name varchar(255) NOT NULL,
product_price varchar(6) NOT NULL,
available_size varchar(5) NOT NULL);
